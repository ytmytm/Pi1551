// This option displays TCBM bus activity on the bottom of the HDMI screen
GraphIEC = 0

// Head-step sound on GPIO 13 (Pi1551 hat SOUND pin). NOT HDMI audio.
//   SoundOnGPIO = 0 (default): legacy Pi1541 — short PWM/DMA sample (PlaySoundDMA) on GPIO13 in ALT0 mode.
//   SoundOnGPIO = 1: square-wave buzzer on GPIO13 (for piezo without built-in generator). Uses Duration/Freq below.
// On Pi1551 hardware with a buzzer on GPIO13, use SoundOnGPIO = 1.
SoundOnGPIO = 1
SoundOnGPIODuration = 500 // Buzz length in milliseconds (only when SoundOnGPIO = 1)
SoundOnGPIOFreq = 1200 // Buzz frequency in Hz (only when SoundOnGPIO = 1)

ShowOptions = 1	// Display parsed options on HDMI startup screen

// Ignore host RESET line: 0 = honour reset (default), 1 = ignore reset pulses
IgnoreReset = 0

//ROTARY:
//
// KY-040 Rotary Encoder Support
//
//    If you would like to use a KY-040 Rotary Encoder for browse menu up/down
//    and select, you can enable it here.  On Pi1551 hat, wiring is fixed:
//
//       GPIO 17  -  Encoder pin A (CLK)  -  hat SW3
//       GPIO 22  -  Encoder pin B (DT)   -  hat SW2
//       GPIO 27  -  Encoder pushbutton   -  hat SW1
//
//    RotaryEncoderInvert flips direction for encoders that turn the wrong way.
//
//    Hold rotary SW1 for 2s to open the ROM selector (browse or emulation).
//    Release, turn to highlight a ROM, then press and release SW1 to apply (emulation: CPU reset + fast boot).
//
RotaryEncoderEnable = 1
//RotaryEncoderInvert = 1

// Display the temperature of the Pi's CPU on HDMI/OLED.
// Should be about 52 C; above 65 C suggests a cooling or power problem.
DisplayTemperature = 1

// Display emulated 6502 program counter on HDMI (debug)
DisplayPC = 0

// Tape motor control option
// If tapeMotorAlwaysOn = 1 (default), motor is always active and tape playback starts immediately
//   this is meant for computers with 6510 CPU replacement, which can't control MOTOR line directly
// If tapeMotorAlwaysOn = 0, motor state is read from TAPE_MOTOR_GPIO pin (GPIO6, physical pin 31)
//   GPIO logic: 0 (LOW) = motor running, 1 (HIGH) = motor stopped
tapeMotorAlwaysOn = 1

// Skip drive motor spin-up delay for 1551: 1 = skip delay, 0 = normal delay
skipMotorSpinUpDelay = 1
